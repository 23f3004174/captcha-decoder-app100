# This file is a complete Flask web app even though it's named index.html. # Run it with: python index.html import os import io import base64 from flask import Flask, request, render_template_string, flash from werkzeug.utils import secure_filename from PIL import Image, ImageOps import pytesseract from pytesseract import TesseractNotFoundError app = Flask(__name__) app.secret_key = os.environ.get("SECRET_KEY", "dev_secret_for_demo") ALLOWED_EXTENSIONS = {"png", "jpg", "jpeg", "gif", "bmp", "tif", "tiff"} TEMPLATE = """ CAPTCHA OCR Decoder

{% endif %}{% endfor %}
{% endif %} {% endwith %}

Detected Text

{{ text if text is not none else "â€”" }}
Tip: For best results, use clear CAPTCHA images with high contrast. This demo applies grayscale + thresholding and a character whitelist.
""" def allowed_file(filename: str) -> bool: return "." in filename and filename.rsplit(".", 1)[1].lower() in ALLOWED_EXTENSIONS def guess_mime(filename: str) -> str: ext = filename.rsplit(".", 1)[1].lower() if "." in filename else "" return { "png": "image/png", "jpg": "image/jpeg", "jpeg": "image/jpeg", "gif": "image/gif", "bmp": "image/bmp", "tif": "image/tiff", "tiff": "image/tiff", }.get(ext, "application/octet-stream") def preprocess_for_captcha(img: Image.Image) -> Image.Image: # Convert to grayscale, then apply simple threshold; this often helps for CAPTCHAs gray = img.convert("L") # Optional normalization: invert if the image seems mostly dark background # Compute mean; if very dark, invert to make foreground darker on light background mean = sum(gray.getdata()) / (gray.width * gray.height) if mean < 100: gray = ImageOps.invert(gray) # Binarize bw = gray.point(lambda x: 0 if x < 150 else 255, "1") return bw @app.route("/", methods=["GET", "POST"]) def index(): decoded_text = None preview_data_url = None if request.method == "POST": if "image" not in request.files: flash("No file part in the request.") return render_template_string(TEMPLATE, text=decoded_text, preview=preview_data_url) file = request.files["image"] if file.filename == "": flash("No file selected.") return render_template_string(TEMPLATE, text=decoded_text, preview=preview_data_url) filename = secure_filename(file.filename) if not allowed_file(filename): flash(f"Unsupported file type. Allowed: {', '.join(sorted(ALLOWED_EXTENSIONS))}") return render_template_string(TEMPLATE, text=decoded_text, preview=preview_data_url) # Read bytes once for both preview and processing data = file.read() if not data: flash("Empty file uploaded.") return render_template_string(TEMPLATE, text=decoded_text, preview=preview_data_url) # Build data URL preview b64 = base64.b64encode(data).decode("ascii") preview_data_url = f"data:{guess_mime(filename)};base64,{b64}" # Open with PIL try: img = Image.open(io.BytesIO(data)) except Exception as e: flash(f"Could not read image: {e}") return render_template_string(TEMPLATE, text=decoded_text, preview=preview_data_url) # Preprocess and OCR try: processed = preprocess_for_captcha(img) # OCR config: single line (psm 7) and alphanumeric whitelist config = "--psm 7 -c tessedit_char_whitelist=ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789" decoded_text = pytesseract.image_to_string(processed, config=config).strip() if not decoded_text: # Try a slightly different page segmentation as fallback fallback_cfg = "--psm 8 -c tessedit_char_whitelist=ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789" decoded_text = pytesseract.image_to_string(processed, config=fallback_cfg).strip() if not decoded_text: flash("OCR completed, but no text was detected. Try a clearer image.") except TesseractNotFoundError: flash("Tesseract OCR engine not found. Please install it and ensure it's on your PATH.") except Exception as e: flash(f"OCR failed: {e}") return render_template_string(TEMPLATE, text=decoded_text, preview=preview_data_url) if __name__ == "__main__": # Optional: let user specify tesseract path via env var TESSERACT_CMD tess_cmd = os.environ.get("TESSERACT_CMD") if tess_cmd: pytesseract.pytesseract.tesseract_cmd = tess_cmd app.run(host="0.0.0.0", port=int(os.environ.get("PORT", 5000)), debug=True)
